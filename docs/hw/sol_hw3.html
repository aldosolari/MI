Exercise 1

  1. Threshold for Higher Criticism
rm(list=ls())
threshold_hc = function(m){
ps = sort(runif(m))
hc = sqrt(m) * ((1:m)/m - ps) / sqrt(ps*(1-ps))
i_max = which.max(hc[1:(m/2)])
hc_max = hc[i_max]
return(c(i_max,hc_max))
}
B = 1000
res = replicate(B,threshold_hc(m=10^6))
summary(res[1,])

  1. Power of Higher Criticism
betas = seq(0.5,1,0.05)
power_hc = function(m,beta){
  rho = ifelse(beta<=0.75, beta-1/2, (1-sqrt(1-beta))^2)
  r = 1.2*rho + 0.1
  z = rnorm(m)
  z[1:round(m^(1-beta))] <- z[1:round(m^(1-beta))] + sqrt(2*r*log(m))
  ps = sort(pnorm(z, lower.tail = F))
  hc = sqrt(m) * ((1:m)/m - ps) / sqrt(ps*(1-ps))
  test = max(hc[1:(m/2)]) > crit
  return(test)
}
B = 100
res = sapply(betas, function(beta) 
  replicate(B, power_hc(m=10^6, beta=beta)) 
  )
plot(betas, apply(res,2,mean), ylab="power", xlab=expression(beta), ylim=c(0,1), type="b")

Exercize 5

Let \(M=\{1,\ldots,m\}\) be the indexes of the null hypotheses \(H_i\) with \(p\)-values \(p_i\), \(i\in M\). Assume \(p_1,\ldots,p_m \stackrel{i.i.d}{\sim}U(0,1)\). For any \(\alpha \in (0,1]\), let \(S=\{i\in M: p_i \leq \alpha \}\) (with \(S=\emptyset\) if \(p_i> \alpha\) for all \(i \in M\)). Note that \(|S| \sim \mathrm{Binomial}(m,\alpha)\). Let \(R = \{i \in S: p_i \leq \alpha/|S|\}\) with \(R=\emptyset\) if \(S =\emptyset\). Define \[\begin{align*} \mathrm{FWER} = & \sum_{k=1}^{m}\mathrm{P}( \cup_{i\in S} \{p_i \leq \alpha/k\}| |S|=k)\mathrm{P}(|S|=k)\\ = & \sum_{k=1}^{m} (1-(1-1/k)^k){ m \choose k }\alpha^k (1-\alpha)^{(m-k)} \end{align*}\]

ms = 1:1000
alpha = 0.05
FWER = sapply(ms, function(m)
    sum(
      sapply(1:m, function(k)
      (1-(1-1/k)^k)*dbinom(k,size=m,prob=alpha)
      )
    )
  )
plot(ms,FWER, type="l", ylim=c(0,1), xlab="m")
