Function to compute the estimated error as a function of \(\gamma\)

hatErr1 = function(gamma, n, m){
  muA = rep(gamma/sqrt(m),m)/2
  muB = -muA
  XA = mvrnorm(n,muA, Sigma = diag(rep(1,m)))
  XB = mvrnorm(n,muB, Sigma = diag(rep(1,m)))
  muAhat = colMeans(XA)
  muBhat = colMeans(XB)
  SD = sqrt( t(muAhat - muBhat)%*%(muAhat - muBhat) )
  PA = pnorm(0, crossprod(muAhat - muBhat, muA - (muAhat + muBhat)/2), sd = SD )
  PB = 1-pnorm(0, crossprod(muAhat - muBhat, muB - (muAhat + muBhat)/2), sd = SD )
  hatErr = PA/2 + PB/2
  return(hatErr)
}

Function to compute the estimated error with the thresholded means

hatErr3 = function(gamma, n, m, s){
  mu = rep(0,m)
  mu[1:s]<- gamma/sqrt(s)
  muA = mu/2
  muB = -mu/2
  XA = mvrnorm(n,muA, Sigma = diag(rep(1,m)))
  XB = mvrnorm(n,muB, Sigma = diag(rep(1,m)))
  muAhat = colMeans(XA)
  muBhat = colMeans(XB)
  lambda <- sqrt(2*log(m)/n)
  muAhat[abs(muAhat) < lambda]<-0
  muBhat[abs(muBhat) < lambda]<-0
  SD = sqrt( t(muAhat - muBhat)%*%(muAhat - muBhat) )
  PA = pnorm(0, crossprod(muAhat - muBhat, muA - (muAhat + muBhat)/2), sd = SD )
  PB = 1-pnorm(0, crossprod(muAhat - muBhat, muB - (muAhat + muBhat)/2), sd = SD )
  hatErr = PA/2 + PB/2
  return(hatErr)
}

rm(list=ls())
m = 10^6
alpha = 0.05
sim = function(m,mu){
  y = rnorm(m,mean=mu)
  p = pnorm(y, lower.tail = F)
  Tmax = max(y) >= qnorm(alpha/m, lower.tail = F)
  Tsum = sum(y)/sqrt(m) >= qnorm(alpha, lower.tail = F)
  Tfisher = -2*sum(log(p)) >= qchisq(alpha, df=2*m, lower.tail = F)
  return(c(Tmax, Tsum, Tfisher))
}
B = 1000
sparse = replicate(B, sim(10^6,mu=c(rep(5,4),rep(0,10^6-4))))
row.names(sparse) = c("Tmax","Tsum","Tfisher")
apply(sparse,1,mean)
##    Tmax    Tsum Tfisher 
##   0.866   0.059   0.058
dense = replicate(B, sim(10^6,mu=c(rep(1.1,2400),rep(0,10^6-2400))))
