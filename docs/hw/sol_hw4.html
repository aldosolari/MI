Exercize 1

Let \(M=\{1,\ldots,m\}\) be the indexes of the null hypotheses \(H_i\) with \(p\)-values \(p_i\), \(i\in M\). Assume \(p_1,\ldots,p_m \stackrel{i.i.d}{\sim}U(0,1)\). For any \(\alpha \in (0,1]\), let \(S=\{i\in M: p_i \leq \alpha \}\) (with \(S=\emptyset\) if \(p_i> \alpha\) for all \(i \in M\)). Let \(R\) be the indexes of the rejections of the Benjamini-Hochberg procedure at level \(\alpha\) applied to the selected hypotheses \(H_i, i \in S\). Note that \(R=S\) and \(|S| \sim \mathrm{Binomial}(m,\alpha)\). Then \[\begin{align*} \mathrm{FDR} = & \mathrm{P}(S \neq \emptyset)\\ = & 1 - (1-\alpha)^m \end{align*}\]

ms = 1:1000
alpha = 0.05
FDR = 1 - (1-alpha)^ms
plot(ms,FDR, type="l", ylim=c(0,1), xlab="m")

Exercize 2

sim = function(m,pi0,rho,alpha){
  z = rnorm(m+1)
  y = z[1]*sqrt(rho) + z[-1]*sqrt(1-rho)
  m1 <- m*(1-pi0)
  if (pi0!=1){ 
    y[1:m1]<- y[1:m1] + rep(1:4, length=m1)
  }
  p = pnorm(y, lower.tail = F)
  hatpi0 = (sum(p > 0.5) + 1)/(0.5*m)
  R_bh = p.adjust(p,"BH") <= alpha
  R_obh = p.adjust(p,"BH") <= (alpha/pi0)
  R_sto = p.adjust(p,"BH") <= (alpha/hatpi0)
  FDR_bh = ifelse(sum(R_bh)>0,sum(R_bh[(m1+1):m])/sum(R_bh), 0)
  FDR_obh = ifelse(sum(R_obh)>0,sum(R_obh[(m1+1):m])/sum(R_obh), 0) 
  FDR_sto = ifelse(sum(R_sto)>0,sum(R_sto[(m1+1):m])/sum(R_sto), 0)
  POW_bh = ifelse(pi0==1,0,sum(R_bh[1:m1])/m1)
  POW_obh = ifelse(pi0==1,0,sum(R_obh[1:m1])/m1)
  POW_sto = ifelse(pi0==1,0,sum(R_sto[1:m1])/m1)
  return(c(FDR_bh,FDR_obh,FDR_sto,POW_bh,POW_obh, POW_sto))
}
